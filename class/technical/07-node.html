            let param = url.parse(req.url,true).query;

            let pdata = '';
            req.on('data',(chunk)=>{
                pdata += chunk;
            });
            req.on('end',()=>{
                let obj = querystring.parse(pdata);
                if(obj.username == 'admin' && obj.password == '123'){
                    res.end('post success');
                }else{
                    res.end('post failure');
                }
            });
        }
    }

}).listen(3000,()=>{

核心index.js

const http = require('http');	//node请求四个方法1
const path = require('path');	//2
const fs = require('fs');		//3
const querystring = require('querystring');	//querystring查询字符串方法4

const scoreData = require('./scores.json');//json数据

http.createServer((req,res)=>{	  // 路由（请求路径+请求方式）
  
  // 查询成绩的入口地址是否正确 /query为入口
  //startsWith()方法 ES6字符串是否在头部  method表单 html中返回表单
  if(req.url.startsWith('/query') && req.method == 'GET'){//判断入口和method表单
     //连接路径 
     fs.readFile(path.join(__dirname,'view','index.tpl'),'utf8',(err,content)=>{
        if(err){	//提示错误信息
          res.writeHead(500,{ 'Content-Type':'text/plain; charset=utf8' });
          res.end('服务器错误，请与管理员联系');
        }
            res.end(content);
      });
  
  //否post 获取成绩的结果 /score
  }else if(req.url.startsWith('/score') && req.method == 'POST'){//是否结果和mathod  
   let pdata = ''; // 定义了一个post变量，用于暂存请求体的信息
   req.on('data',(chunk)=>{// 通过req的data事件监听函数，每当接受到请求体的数据，就累加到post变量中
     pdata += chunk;
   });  
   //在end事件触发后，通过querystring.parse将post解析为真正的POST请求格式，然后向客户端返回
   req.on('end',()=>{	//Jquery 原始为 ,function(){    
     //解析参数
     let obj = querystring.parse(pdata);
     let result = scoreData[obj.code];
     //连接路径
     fs.readFile(path.join(__dirname,'view','result.tpl'),'utf8',(err,content)=>{
       if(err){ //提示错误信息
         res.writeHead(500,{ 'Content-Type':'text/plain; charset=utf8' });
        res.end('服务器错误，请与管理员联系');
      }
        
      // 返回内容 之前要进行数据渲染
      content = content.replace('$$chinese$$',result.chinese);
      content = content.replace('$$math$$',result.math);
      content = content.replace('$$english$$',result.english);
      content = content.replace('$$summary$$',result.summary);
      res.end(content);//返回数据
     });
   });
 }
}).listen(3000,()=>{	//监听端口3000
    console.log('running....'); //（可省略）
});

const mysql = require('mysql');	//创建请求sql数据

exports.base = (sql,data,callback) => {
    // 创建数据库连接
    const connection = mysql.createConnection({
        host: 'localhost', // 数据库所在的服务器的域名或者IP地址
        user: 'root', // 登录数据库的账号
        password: '', // 登录数据库的密码
        database: 'book' // 数据库名称
    });
    // 执行连接操作
    connection.connect();

    // 操作数据库(数据库操作也是异步的)
    connection.query(sql,data, function(error, results, fields) 	{
        if (error) throw error;//报错提示
        callback(results);
    });
    // 关闭数据库
    connection.end();
}

const data = require('./data.json');

// 自动生成图书编号（自增）
let maxBookCode = ()=>{
    let arr = [];
    data.forEach((item)=>{
        arr.push(item.id);
    });
    return Math.max.apply(null,arr);
}

// 渲染主页面
exports.showIndex = (req,res) => {	
    let sql = 'select * from book';	//sql查询
    db.base(sql,null,(result)=>{	//查询数据返回
        res.render('index',{list : result});
    });
}

// 跳转到添加图书的页面
exports.toAddBook = (req,res) => {
    res.render('addBook',{});
}
// 添加图书保存数据
exports.addBook = (req,res) => {
    // 获取表单数据
    let info = req.body;
    let book = {};
    for(let key in info){	//添加数据
        book[key] = info[key];
    }
    let sql = 'insert into book set ?';	//sql数据添加
    db.base(sql,book,(result)=>{	//返回添加数据
        if(result.affectedRows == 1){	//1
            res.redirect('/');
        }
    });
}

// 跳转编辑图书页面
exports.toEditBook = (req,res) => {
    let id = req.query.id;
    let sql = 'select * from book where id=?';//sql查询
    let data = [id];
    db.base(sql,data,(result)=>{
        res.render('editBook',result[0]);
    });
}
// 编辑图书更新数据
exports.editBook = (req,res) => {
    let info = req.body;
    let sql = 'update book set name=?,author=?,category=?,description=? where id=?';//sql编辑
    let data = [info.name,info.author,info.category,info.description,info.id];
    db.base(sql,data,(result)=>{
        if(result.affectedRows == 1){
            res.redirect('/');
        }
    });
}
// 删除图书信息
exports.deleteBook = (req,res) => {
    let id = req.query.id;
    let sql = 'delete from book where id=?';	//sql删除
    let data = [id];
    db.base(sql,data,(result)=>{
        if(result.affectedRows == 1){
            res.redirect('/');
        }
    });
}
